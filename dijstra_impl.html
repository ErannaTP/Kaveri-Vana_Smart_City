AVL Tree Surveillance Events

The following is a C++ implementation for managing and querying events within an AVL tree.

Code Implementation

#include <iostream>
#include <algorithm>
using namespace std;

// AVL Tree Node structure
struct AVLNode {
    int timestamp;
    string eventDetails;
    int height;
    AVLNode* left;
    AVLNode* right;
};

// Function to get the height of the node
int getHeight(AVLNode* node) {
    if (node == nullptr) return 0;
    return node->height;
}

// Function to get the balance factor of a node
int getBalance(AVLNode* node) {
    if (node == nullptr) return 0;
    return getHeight(node->left) - getHeight(node->right);
}

// Right Rotation
AVLNode* rightRotate(AVLNode* y) {
    AVLNode* x = y->left;
    AVLNode* T2 = x->right;

    x->right = y;
    y->left = T2;

    y->height = max(getHeight(y->left), getHeight(y->right)) + 1;
    x->height = max(getHeight(x->left), getHeight(x->right)) + 1;

    return x;
}

// Left Rotation
AVLNode* leftRotate(AVLNode* x) {
    AVLNode* y = x->right;
    AVLNode* T2 = y->left;

    y->left = x;
    x->right = T2;

    x->height = max(getHeight(x->left), getHeight(x->right)) + 1;
    y->height = max(getHeight(y->left), getHeight(y->right)) + 1;

    return y;
}

// Function to create a new AVL Node
AVLNode* newAVLNode(int timestamp, string eventDetails) {
    AVLNode* node = new AVLNode();
    node->timestamp = timestamp;
    node->eventDetails = eventDetails;
    node->height = 1;
    node->left = node->right = nullptr;
    return node;
}

// Function to insert a node into AVL Tree
AVLNode* insertAVL(AVLNode* node, int timestamp, string eventDetails) {
    if (node == nullptr) return newAVLNode(timestamp, eventDetails);

    if (timestamp < node->timestamp) {
        node->left = insertAVL(node->left, timestamp, eventDetails);
    } else if (timestamp > node->timestamp) {
        node->right = insertAVL(node->right, timestamp, eventDetails);
    } else {
        return node; // Duplicate timestamps are not allowed
    }

    node->height = 1 + max(getHeight(node->left), getHeight(node->right));

    int balance = getBalance(node);

    // Left Left Case
    if (balance > 1 && timestamp < node->left->timestamp) {
        return rightRotate(node);
    }

    // Right Right Case
    if (balance < -1 && timestamp > node->right->timestamp) {
        return leftRotate(node);
    }

    // Left Right Case
    if (balance > 1 && timestamp > node->left->timestamp) {
        node->left = leftRotate(node->left);
        return rightRotate(node);
    }

    // Right Left Case
    if (balance < -1 && timestamp < node->right->timestamp) {
        node->right = rightRotate(node->right);
        return leftRotate(node);
    }

    return node;
}

// Function to search events in a given range
void queryAVL(AVLNode* root, int start, int end) {
    if (root == nullptr) return;

    if (start < root->timestamp) {
        queryAVL(root->left, start, end);
    }

    if (start <= root->timestamp && end >= root->timestamp) {
        cout << "Timestamp: " << root->timestamp << ", Event: " << root->eventDetails << endl;
    }

    if (end > root->timestamp) {
        queryAVL(root->right, start, end);
    }
}

int main() {
    AVLNode* root = nullptr;

    // Adding surveillance events
    root = insertAVL(root, 1615309200, "Suspicious activity at Gate 1");
    root = insertAVL(root, 1615312800, "Traffic congestion at Main Square");
    root = insertAVL(root, 1615316400, "Fire reported near Park Area");
    root = insertAVL(root, 1615320000, "Unauthorized access to Restricted Zone");
    root = insertAVL(root, 1615323600, "Earthquake warning issued");

    cout << "Surveillance Events between timestamps 1615309200 and 1615317000:\n";
    queryAVL(root, 1615309200, 1615317000);

    return 0;
}
