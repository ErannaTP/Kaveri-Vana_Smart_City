Tourism Trie Exploration

The following is a C++ implementation for managing tourism recommendations using a Trie data structure.

Code Implementation

#include <iostream>
#include <unordered_map>
#include <vector>
using namespace std;

struct TrieNode {
    unordered_map children;
    vector places;
};

class Trie {
private:
    TrieNode* root;

    TrieNode* createNode() {
        return new TrieNode();
    }

public:
    Trie() {
        root = createNode();
    }

    void insert(const string& keyword, const string& place) {
        TrieNode* current = root;
        for (char ch : keyword) {
            if (!current->children[ch]) {
                current->children[ch] = createNode();
            }
            current = current->children[ch];
        }
        current->places.push_back(place);
    }

    vector search(const string& keyword) {
        TrieNode* current = root;
        for (char ch : keyword) {
            if (!current->children[ch]) {
                return {}; // No matches
            }
            current = current->children[ch];
        }
        return current->places;
    }
};

int main() {
    Trie tourismTrie;

    // Adding tourist spots near Mysore and Bangalore
    tourismTrie.insert("heritage", "Mysore Palace");
    tourismTrie.insert("heritage", "Tipu Sultan's Summer Palace, Srirangapatna");
    tourismTrie.insert("wildlife", "Mysore Zoo");
    tourismTrie.insert("wildlife", "Ranganathittu Bird Sanctuary");
    tourismTrie.insert("spiritual", "Chamundi Hills Temple");
    tourismTrie.insert("spiritual", "St. Philomena's Church");
    tourismTrie.insert("nature", "Brindavan Gardens");
    tourismTrie.insert("nature", "KRS Dam");
    tourismTrie.insert("heritage", "Bangalore Palace");
    tourismTrie.insert("heritage", "Lalbagh Botanical Garden, Bangalore");
    tourismTrie.insert("wildlife", "Bannerghatta National Park");
    tourismTrie.insert("spiritual", "ISKCON Temple, Bangalore");

    string query;
    cout << "Enter a category to explore (e.g., heritage, wildlife, spiritual, nature): ";
    cin >> query;

    vector recommendations = tourismTrie.search(query);
    if (!recommendations.empty()) {
        cout << "Recommended places for " << query << ":\n";
        for (const string& place : recommendations) {
            cout << "- " << place << endl;
        }
    } else {
        cout << "No recommendations found for the category: " << query << endl;
    }

    return 0;
}
